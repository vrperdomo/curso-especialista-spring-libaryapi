Check port:

netstat -aof | findstr 5432

# Criando uma network
docker network create library-network

# Comando para criar um container do PostgreSQL
docker run --name libarydb -p 5432:5432 -e POSTGRES_PASSWORD=postgres -e POSTGRES_USER=postgres -e POSTGRES_DB=libary --network libary-network postgres:18.3

# Comando para criar um container do pgadmin4
docker run --name pgadmin4 -e PGADMIN_DEFAULT_EMAIL=admin@admin.com -e PGADMIN_DEFAULT_PASSWORD=admin -p 15432:80 -d --network library-network dpage/pgadmin4
